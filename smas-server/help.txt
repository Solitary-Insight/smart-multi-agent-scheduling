SELECT CONCAT('DESCRIBE ', table_name, ';')
FROM information_schema.tables
WHERE table_schema = 'SMAS';


//INFO: QUICK VERIFICATION OF CREDIT HOURS     
SELECT 
    c.course_name,
    c.credit_hours,
    COUNT(t.id) AS scheduled_slots,
    case 
    when c.credit_hours=scheduled_slots then "VALID"
    when c.credit_hours<scheduled_slots then "LESS SLOTS"
    when c.credit_hours>scheduled_slots then "MORE SLOTS" as slot_distributions 
FROM courses c
LEFT JOIN timetable_slots t 
    ON t.course_id = c.id 
    AND t.timetable_id = (SELECT MAX(id) FROM timetables)
GROUP BY c.id, c.course_name, c.credit_hours
ORDER BY c.id;
